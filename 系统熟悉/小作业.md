## 数据库表设计

### CPC表的设计
SQL代码：
``` sql
  CREATE DATABASE /*!32312 IF NOT EXISTS*/`demo_for_ad` /*!40100 DEFAULT CHARACTER SET utf8 */;
 USE `demo_for_ad`;
 DROP TABLE IF EXISTS `cpc`;

CREATE TABLE `cpc` (
   `CPC_ID` BIGINT(20) NOT NULL AUTO_INCREMENT COMMENT 'ID',
   `ACCOUNT_ID` INT(10) NOT NULL COMMENT '账号ID',
   `KEY` VARCHAR(64) NOT NULL COMMENT '投放的关键词',
   `URL` varchar(1024) DEFAULT NULL COMMENT '广告url',
	 `CREATE_DATE` DATETIME  NOT NULL DEFAULT NOW() COMMENT '广告创建时间',
	 `CHG_DATE` DATETIME  NOT null DEFAULT NOW() COMMENT '广告修改时间',
	 `MAX_PRICE` INT(6) NOT NULL DEFAULT 30 COMMENT '最高点击价格,单位为分',
   PRIMARY KEY (`CPC_ID`)
 ) ENGINE=INNODB AUTO_INCREMENT=1029 DEFAULT CHARSET=utf8;
```
